Function Cedula(J2 As Variant) As Variant
    Dim resultado As Variant
    resultado = Application.VLookup(J2, Sheets("Hoja1").Range("A:B"), 1, False)
    If IsError(resultado) Then
        Cedula = ""
    Else
        Cedula = resultado
    End If
End Function


Function Profesional(J2 As Variant) As Variant

End Function
